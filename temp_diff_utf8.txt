+def process_and_deduplicate_transactions(transactions):
+    if not transactions:
+        return []
+
+    # Sort chronologically ascending
+    transactions = sorted(transactions, key=lambda x: x.timestamp or datetime.min)
+
+    seen = set()
+    deduped = []
+    for tx in transactions:
+        tx_code = tx.transaction_code or ''
+        amt = tx.amount
+        ts_str = tx.timestamp.strftime('%Y-%m-%d %H:%M:%S') if tx.timestamp else ''
+        key = (tx_code, amt, ts_str)
+        if key not in seen:
+            seen.add(key)
+            deduped.append(tx)
+
+    # Recompute running balance forward
+    if deduped:
+        running_balance = deduped[0].balance
+        for i in range(1, len(deduped)):
+            tx = deduped[i]
+            if tx.category == 'inflow':
+                running_balance += tx.amount
+            else:
+                running_balance -= tx.amount
+            tx.balance = running_balance
+
+    return deduped
